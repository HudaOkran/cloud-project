# أوامر الإعداد والتكوين
git config --global user.name "اسمك"
git config --global user.email "بريدك@example.com"
git config --global init.default branch main
git config --list

# أوامر إدارة المستودعات
git init
git status
git add <اسم الملف>
git add --all

# أوامر حفظ التغييرات
git commit -m "رسالة وصفية"
git log
git log --oneline

# أوامر الفروع
git branch
git branch <اسم الفرع>
git switch <اسم الفرع>
git checkout <اسم الفرع>
git merge <اسم الفرع>
git branch -d <اسم الفرع>

# أوامر رفع وسحب التغييرات
git remote add origin <رابط المستودع>
git push -u origin main
git push origin main --force
git pull origin main

# أوامر المساعدة والاستعادة
git diff
git restore <اسم الملف>
git help <أمر>
clear

# أوامر التنقل
cd <اسم المجلد>
