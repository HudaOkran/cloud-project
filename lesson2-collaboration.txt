# أوامر التعاون (Collaboration) على GitHub

# نسخ مستودع من GitHub إلى الجهاز
git clone <رابط المستودع>

# ربط المستودع المحلي بمستودع على GitHub
git remote add origin <رابط المستودع>

# عرض الروابط المرتبطة بالمستودع
git remote -v

# إلغاء ربط المستودع بـ GitHub
git remote remove origin

# سحب آخر التغييرات من GitHub
git pull origin main

# رفع التغييرات إلى GitHub
git push origin main

# رفع التغييرات بالقوة (استخدم بحذر)
git push origin main --force

# تغيير اسم الفرع الرئيسي
git branch -M main

# عرض جميع الفروع
git branch

# إنشاء فرع جديد
git branch <اسم الفرع>

# الانتقال إلى فرع
git switch <اسم الفرع>
